Use wowapi;

CREATE TABLE WOWAccount ( accountId INT UNSIGNED PRIMARY KEY AUTO_INCREMENT NOT NULL, accountName VARCHAR(255) NOT NULL, link VARCHAR(255) NOT NULL, isActive BOOLEAN DEFAULT TRUE);
